SELECT "PetID"
  ,"PetName"
  ,"Species"
  ,"Breed"
  ,"Size"
  ,"Gender"
  ,"Birthdate"
  ,"PetDescription"
  ,"Status"
  ,p."ProfileID"
  ,"ProfileName"
  ,"ProfileDescription"
  ,"Address"
  ,"City"
  ,"State"
  ,"Zip"
  ,"Phone"
  ,"ProfileEmail"
FROM "Pet" p
JOIN "Profile" pr ON p."ProfileID" = pr."ProfileID"
WHERE "Status" = 'available'
ORDER BY "ProfileName", "PetName"
