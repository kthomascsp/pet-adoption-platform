SELECT p."PetID"
  ,"PetName"
  ,"Species"
  ,"Breed"
  ,"Birthdate"
  ,"Size"
  ,"Gender"
  ,"PetDescription"
  ,"Status"
  ,p."ProfileID"
  ,"ProfileType"
  ,"ProfileName"
  ,"Address"
  ,"City"
  ,"State"
  ,"Zip"
  ,"Phone"
  ,"ProfileDescription"
  ,pi."ImageID"
  ,"URL"
  ,"ImageDescription"
FROM "Pet" p
JOIN "Profile" pr ON p."ProfileID" = pr."ProfileID"
JOIN "PetImage" pi ON p."PetID" = pi."PetID"
JOIN "Image" i ON pi."ImageID" = i."ImageID"
