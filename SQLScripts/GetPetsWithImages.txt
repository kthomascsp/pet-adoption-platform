SELECT p."PetID"
  ,p."Name" AS "PetName"
  ,"Species"
  ,"Breed"
  ,"Birthdate"
  ,"Size"
  ,"Gender"
  ,p."Description" AS "PetDescription"
  ,"Status"
  ,p."ProfileID"
  ,"Type"
  ,pr."Name" AS "ShelterName"
  ,"Address"
  ,"City"
  ,"State"
  ,"Zip"
  ,"Phone"
  ,pr."Description"
  ,pr."ImageID"
  ,"URL"
  ,i."Description"
  ,"Sequence"
FROM "Pet" p
JOIN "Profile" pr ON p."ProfileID" = pr."ProfileID"
JOIN "PetImage" pi ON p."PetID" = pi."PetID"
JOIN "Image" i ON pi."ImageID" = i."ImageID"
